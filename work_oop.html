OOP Game Development

A group project creating an interactive game using object-oriented principles in Java, with focus on movement logic for entities.

Project Description

This group project involved developing a game where tasks were divided among team members: scene management, movement, collision detection, I/O handling, and entity management. My role was implementing the movement logic, including AI-controlled entity movement, player-controlled movements with boundaries, and physics-based effects like gravity and swaying for realism. The game uses OOP principles such as interfaces (e.g., iMoveable), abstraction, and encapsulation to manage entity behaviors efficiently.

package MovementManager;

import java.util.ArrayList;
import java.util.List;
import EntityManager.*;

public class MovementManager {
    private List movables;

    // Global gravity settings (controls fall behavior)
    private float gravityAcceleration = 0.1f;
    private float maxFallSpeed = 4f;

    // Time counter for smooth oscillation/sway effect
    private float timeCounter = 0;

    public MovementManager() {
        this.movables = new ArrayList<>();
    }

    // Register a moveable entity (e.g., Food, Paper, Plastic, Player)
    public void addMovable(iMoveable entity) {
        movables.add(entity);
    }

    // Remove a moveable entity
    public void removeMovable(iMoveable entity) {
        movables.remove(entity);
    }

    // Main update loop called every frame
    public void update(float deltaTime) {
        timeCounter += deltaTime; // Keep track of time for oscillation

        for (iMoveable entity : movables) {
            if (entity instanceof Entity) {
                applyGravity((Entity) entity);       // Apply gravity (falling effect)
                applySway((Entity) entity, timeCounter); // Apply side-to-side swaying if needed
            }
            if (entity instanceof TextureObject && ((TextureObject) entity).getType().equals("Recycling_Bin")) {
                applySmoothPlayerMovement((TextureObject) entity); // Smooth bucket control
            }
            applyAI(entity); // Execute AI-based movement (e.g., falling down)
            applyPlayerMovement(entity, null); // Handle player input (for player-controlled entities)
        }
    }

    // Call AI movement logic defined inside each entity
    public void applyAI(iMoveable entity) {
        entity.movementAI();
    }

    // Call player-controlled movement logic
    public void applyPlayerMovement(iMoveable entity, String direction) {
        entity.movementPlayer();
    }

    // Apply gravity acceleration, capped at max speed, with easing effect
    public void applyGravity(Entity entity) {
        float gravityFactor = getGravityFactor(entity); // Different fall speed for different materials
        float currentSpeed = entity.getSpeed();

        // Accelerate smoothly, don't exceed max speed
        if (currentSpeed < maxFallSpeed * gravityFactor) {
            float easedAcceleration = gravityAcceleration * gravityFactor;
            entity.setSpeed(currentSpeed + easedAcceleration);
        }
    }

    // Assign different gravity scaling factors for each entity type
    private float getGravityFactor(Entity entity) {
        if (entity instanceof TextureObject) {
            TextureObject textureObject = (TextureObject) entity;
            String type = textureObject.getType();
            if (type.equals("Styrofoam")) return 0.4f; // Very slow fall
            if (type.equals("Paper")) return 0.7f;    // Medium fall
        }
        return 1.0f; // Normal fall for other items (e.g., Food, Plastic)
    }

    // Apply different swaying effects for light materials (Paper and Styrofoam)
    private void applySway(Entity entity, float time) {
        if (entity instanceof TextureObject) {
            TextureObject textureObject = (TextureObject) entity;
            String type = textureObject.getType();
            if (type.equals("Styrofoam")) {
                // Wide and slow sway for Styrofoam
                float swayAmount = 1.5f; // Larger sway distance
                float swaySpeed = 1.0f;  // Slower oscillation
                entity.setX(entity.getX() + (float) Math.sin(time * swaySpeed) * swayAmount);
            } else if (type.equals("Paper")) {
                // Narrow and fast sway for Paper
                float swayAmount = 0.8f; // Smaller sway distance
                float swaySpeed = 2.5f;  // Faster oscillation
                entity.setX(entity.getX() + (float) Math.sin(time * swaySpeed) * swayAmount);
            }
        }
    }

    // Smooth, inertia-based player-controlled movement (Recycling Bin)
    private void applySmoothPlayerMovement(TextureObject entity) {
        if (entity.getType().equals("Recycling_Bin")) {
            // Call the smooth movement logic for the Recycling Bin
            entity.movementPlayer(); // Assuming movementPlayer() handles smooth movement
        }
    }
}
