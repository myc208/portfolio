Testimonial

Internship Overview

During this internship, I collaborated with a partner to develop defect visualization systems for three machines: AOI, SPI, and SAOI. We engaged engineers from multiple sites to understand visualization needs for root cause analysis and reporting. The process involved data gathering from sub-teams, iterative testing with engineers, accuracy verification in Singapore, fine-tuning, documentation, and handover to IT with detailed guides. The systems process inspection data, generate interactive plots, and manage storage efficiently.

def create_multi_board_layout(components, defects, side):
    board_count = 5
    points = []
    for board_num in range(1, board_count + 1):
        for comp in components:
            point = {**comp, 'BoardNumber': board_num}
            key = (board_num, comp['LocationName'].upper().strip(), comp['PartNumber'].upper().strip())
            if key in defects:
                point['DefectType'] = defects[key][0] if defects[key] else "NONE"
                point['DefectCount'] = len(defects[key])
            else:
                point['DefectType'] = "NONE"
                point['DefectCount'] = 0
            points.append(point)
    return points

def process_txt_file(file_path):
    components = []
    min_x, min_y = float('inf'), float('inf')
    max_x, max_y = float('-inf'), float('-inf')
    with open(file_path, 'r') as f:
        for line in f:
            parts = line.strip().split()
            if len(parts) >= 6 and parts[0] != "fid":
                location_name = parts[0]
                part_number = parts[1]
                x = float(parts[2])
                y = float(parts[3])
                rotation = float(parts[4])
                package = parts[5]
                components.append({
                    'LocationName': location_name,
                    'PartNumber': part_number,
                    'X': x,
                    'Y': y,
                    'Rotation': rotation,
                    'Package': package
                })
                min_x = min(min_x, x)
                min_y = min(min_y, y)
                max_x = max(max_x, x)
                max_y = max(max_y, y)
    # Normalize coordinates to start from 0,0
    for comp in components:
        comp['X'] -= min_x
        comp['Y'] -= min_y
    board_width = max_x - min_x
    board_height = max_y - min_y
    return components, {'Width': board_width, 'Height': board_height}
