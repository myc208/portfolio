Employee Application Development Using C#

Develop a robust application to better manage various employees of a company.

Project Description

This project is a simple system for managing employees in an organization. It allows users to add employees, including Full-Time, Part-Time, and Sales employees, while also calculating and displaying their pay. Users can search for employees by ID, remove employees by name, and sort them by their pay. The system is built using C# and demonstrates key programming concepts such as organizing data with class hierarchies and handling errors gracefully.

Technologies Used

  • Github
  • Visual Studio Code
  • C#

Summary of Steps

  • Define project requirements
  • Plan the class design
  • Develop the core functionality
  • Add error handling
  • Add logging for easier debugging
  • Debug and test
  • Make iterative refinements before making final testing

Employee.cs


            
            using System;

            namespace EmployeeApp_P11
            {
                abstract class Employee : IComparable
                {
                    public int Id { get; set; }
                    public string Name { get; set; }
            
                    public Employee() { }
                    public Employee(int id, string name)
                    {
                        try
                        {
                            Id = id;
                            Name = name;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error initializing Employee: {ex.Message}");
                        }
                    }
            
                    public abstract double CalculatePay();
            
                    public override string ToString()
                    {
                        try
                        {
                            return $"\tId: {Id}\tName: {Name}";
                        }
                        catch (Exception ex)
                        {
                            return $"Error generating string representation: {ex.Message}";
                        }
                    }
            
                    public int CompareTo(Employee e)
                    {
                        try
                        {
                            if (CalculatePay() > e.CalculatePay())
                                return 1;
                            else if (CalculatePay() == e.CalculatePay())
                                return 0;
                            else
                                return -1;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error comparing employees: {ex.Message}");
                            return 0;
                        }
                    }
                }
            }

FullTimeEmployee.cs


            
            using System;

            namespace EmployeeApp_P11
            {
                class FullTimeEmployee : Employee
                {
                    public double BasicPay { get; set; }
                    public double Allowance { get; set; }
            
                    public FullTimeEmployee() : base() { }
            
                    public FullTimeEmployee(int id, string name, double bp, double alw)
                        : base(id, name)
                    {
                        try
                        {
                            BasicPay = bp;
                            Allowance = alw;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error initializing FullTimeEmployee: {ex.Message}");
                        }
                    }
            
                    public override double CalculatePay()
                    {
                        try
                        {
                            return BasicPay + Allowance;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error calculating pay for FullTimeEmployee: {ex.Message}");
                            return 0;
                        }
                    }
            
                    public override string ToString()
                    {
                        try
                        {
                            return base.ToString() + $"\tBasic Pay: {BasicPay}\tAllowance: {Allowance}";
                        }
                        catch (Exception ex)
                        {
                            return $"Error generating string representation: {ex.Message}";
                        }
                    }
                }
            }

SalesEmployee.cs


            
            using System;

            namespace EmployeeApp_P11
            {
                class SalesEmployee : FullTimeEmployee
                {
                    public double SalesAmount { get; set; }
            
                    public SalesEmployee() : base() { }
            
                    public SalesEmployee(int id, string name, double bp, double alw, double sa)
                        : base(id, name, bp, alw)
                    {
                        try
                        {
                            SalesAmount = sa;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error initializing SalesEmployee: {ex.Message}");
                        }
                    }
            
                    public double DetermineCommissionRate()
                    {
                        try
                        {
                            return SalesAmount < 1000 ? 0.2 : 0.5;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error determining commission rate: {ex.Message}");
                            return 0;
                        }
                    }
            
                    public override double CalculatePay()
                    {
                        try
                        {
                            return base.CalculatePay() + SalesAmount * DetermineCommissionRate();
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error calculating pay for SalesEmployee: {ex.Message}");
                            return 0;
                        }
                    }
            
                    public override string ToString()
                    {
                        try
                        {
                            return base.ToString() + $"\tSales Amount: {SalesAmount}";
                        }
                        catch (Exception ex)
                        {
                            return $"Error generating string representation: {ex.Message}";
                        }
                    }
                }
            }

Program.cs


            
            using System;
            using System.Collections.Generic;
            
            namespace EmployeeApp_P11
            {
                class Program
                {
                    static void Main(string[] args)
                    {
                        List employeeList = new List
                        {
                            new FullTimeEmployee(103, "John", 1500, 100),
                            new PartTimeEmployee(101, "Mary", 50, 10),
                            new SalesEmployee(102, "Apple", 1000, 50, 10000)
                        };
            
                        try
                        {
                            Console.WriteLine("Adding employees...");
                            string type = "";
                            while (type != "Exit")
                            {
                                Console.Write("Enter Employee type (FullTimeEmployee, PartTimeEmployee, SalesEmployee, Exit to stop): ");
                                type = Console.ReadLine();
                                if (type == "Exit")
                                {
                                    break;
                                }
                                else
                                {
                                    Employee e = AddEmployee(type);
                                    if (e != null)
                                    {
                                        employeeList.Add(e);
                                    }
                                }
            
                                Console.WriteLine();
                            }
            
                            DisplayOutput(employeeList);
            
                            Console.WriteLine("\nIncreasing pay...");
                            IncreasePay(employeeList);
                            DisplayOutput(employeeList);
            
                            Console.Write("\nSearch Employee by ID: ");
                            int searchId = Convert.ToInt32(Console.ReadLine());
                            Employee foundEmployee = SearchEmployee(employeeList, searchId);
                            if (foundEmployee != null)
                            {
                                Console.WriteLine("Employee Found: \n{0}", foundEmployee);
                            }
                            else
                            {
                                Console.WriteLine("No employee found with the given ID.");
                            }
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error in Main Method: {ex.Message}");
                        }
            
                        Console.ReadLine();
                    }
            
                    static Employee AddEmployee(string type)
                    {
                        try
                        {
                            Console.Write("Enter ID: ");
                            int id = Convert.ToInt32(Console.ReadLine());
                            Console.Write("Enter Name: ");
                            string name = Console.ReadLine();
            
                            if (type == "PartTimeEmployee")
                            {
                                Console.Write("Daily Rate: ");
                                double dailyRate = Convert.ToDouble(Console.ReadLine());
                                Console.Write("Days Worked: ");
                                int daysWorked = Convert.ToInt32(Console.ReadLine());
            
                                return new PartTimeEmployee(id, name, dailyRate, daysWorked);
                            }
                            else
                            {
                                Console.Write("Basic Pay: ");
                                double basicPay = Convert.ToDouble(Console.ReadLine());
                                Console.Write("Allowance: ");
                                double allowance = Convert.ToDouble(Console.ReadLine());
            
                                if (type == "SalesEmployee")
                                {
                                    Console.Write("Sales Amount: ");
                                    double salesAmount = Convert.ToDouble(Console.ReadLine());
                                    return new SalesEmployee(id, name, basicPay, allowance, salesAmount);
                                }
                                else if (type == "FullTimeEmployee")
                                {
                                    return new FullTimeEmployee(id, name, basicPay, allowance);
                                }
                                else
                                {
                                    Console.WriteLine("Invalid employee type.");
                                    return null;
                                }
                            }
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error adding employee: {ex.Message}");
                            return null;
                        }
                    }
            
                    static void DisplayOutput(List eList)
                    {
                        try
                        {
                            foreach (var e in eList)
                            {
                                Console.WriteLine(e);
                            }
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error displaying employees: {ex.Message}");
                        }
                    }
            
                    static void IncreasePay(List eList)
                    {
                        try
                        {
                            foreach (Employee e in eList)
                            {
                                if (e is FullTimeEmployee f)
                                {
                                    f.BasicPay *= 1.1;
                                }
                            }
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error increasing pay: {ex.Message}");
                        }
                    }
            
                    static Employee SearchEmployee(List eList, int id)
                    {
                        try
                        {
                            foreach (Employee e in eList)
                            {
                                if (e.Id == id)
                                {
                                    return e;
                                }
                            }
                            return null;
                        }
                        catch (Exception ex)
                        {
                            Console.WriteLine($"Error searching for employee: {ex.Message}");
                            return null;
                        }
                    }
                }
            }
