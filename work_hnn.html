# Hybrid Model Architecture
def create_hybrid_model(max_features, embedding_dim, max_length):
    inputs = Input(shape=(max_length,))
    embedding = Embedding(max_features, embedding_dim, input_length=max_length)(inputs)
    
    # CNN Branch
    conv1 = Conv1D(128, 5, activation='relu')(embedding)
    pool1 = MaxPooling1D(5)(conv1)
    conv2 = Conv1D(128, 5, activation='relu')(pool1)
    pool2 = MaxPooling1D(5)(conv2)
    
    # LSTM Branch
    lstm1 = LSTM(64, return_sequences=True)(embedding)
    lstm2 = LSTM(64)(lstm1)
    
    # Merge branches
    merged = concatenate([pool2, lstm2])
    
    # Dense layers
    dense1 = Dense(64, activation='relu')(merged)
    dropout = Dropout(0.5)(dense1)
    output = Dense(1, activation='sigmoid')(dropout)
    
    model = Model(inputs=inputs, outputs=output)
    model.compile(loss='binary_crossentropy',
                 optimizer='adam',
                 metrics=['accuracy'])
    return model
